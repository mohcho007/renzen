function shuffle(arr) {
  let a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

  <h2 class="renzen-faq__title">Ofte stillede spørgsmål (FAQ)</h2>

  <div class="renzen-faq__grid">
    <details class="renzen-faq__item" open>
      <summary class="renzen-faq__q">Hvad er introtilbuddet til 179 kr.?</summary>
      <div class="renzen-faq__a">
        Introtilbuddet gælder din første rengøring. Det er en engangspris, så du kan prøve en Zenmester uden risiko.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Hvad sker der efter første rengøring?</summary>
      <div class="renzen-faq__a">
        Efter din første rengøring kan du vælge at fortsætte som medlem af Renzen Klubben. Som medlem betaler du kun 119 kr./md. og får op til 25% rabat på hver rengøring samt 3.000 kr. i kreditter til andre services.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Er der binding på medlemskabet?</summary>
      <div class="renzen-faq__a">
        Ja, medlemskabet løber i minimum 6 måneder. Efter perioden kan du opsige når som helst.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Hvordan fungerer kreditten på 3.000 kr.?</summary>
      <div class="renzen-faq__a">
        Du får 300 kr. i kredit pr. måned, som kan bruges på 10 udvalgte services (fx flyttehjælp, malerarbejde, affaldshåndtering m.m.). Samlet kan du spare op til 3.000 kr. på et år.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Kan jeg bruge alle 3.000 kr. på én gang?</summary>
      <div class="renzen-faq__a">
        Nej. Kreditten frigives løbende (300 kr./md.), så du kan planlægge og bruge den over tid.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Hvad er inkluderet i en standardrengøring?</summary>
      <div class="renzen-faq__a">
        Standardrengøring dækker støvsugning, gulvvask, aftørring af overflader, køkken og badeværelse. Ekstra ydelser (fx ovnrens, vinduespudsning, møbelrens) kan tilkøbes.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Hvad er ikke inkluderet?</summary>
      <div class="renzen-faq__a">
        Flytning af tunge møbler, udendørsopgaver, tæpperens, skadedyrsbekæmpelse og dyb pletfjerning er ikke inkluderet i standardrengøring.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Hvorfor vælge en Zenmester via Renzen?</summary>
      <div class="renzen-faq__a">
        Alle Zenmestre er verificerede, forsikrede og vurderet af andre kunder. Du får en pålidelig og professionel rengøring – hver gang.
      </div>
    </details>

    <details class="renzen-faq__item">
      <summary class="renzen-faq__q">Kan jeg ændre eller aflyse en rengøring?</summary>
      <div class="renzen-faq__a">

      </div>
