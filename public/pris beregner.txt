<script>
(function() {
  // ✅ RENZEN base pricing model (your current base prices)
  const BASE_FEE = 245;     // fast base
  const PRICE_PER_M2 = 8;   // kr pr m²

  // service mapping (unchanged structure; used for dataset only)
  const sqmTiers = [
    { max: 70,  serviceId: 81 },
    { max: 100, serviceId: 82 },
    { max: 130, serviceId: 83 },
    { max: 160, serviceId: 84 },
    { max: 200, serviceId: 85 },
    { max: 230, serviceId: 86 },
    { max: 260, serviceId: 87 },
    { max: 299, serviceId: 88 }
  ];

  const frequencyMap = [
    { id: 1,  mult: 1,    label: '1-gang',     discount: 0 },
    { id: 17, mult: 0.8,  label: 'hver-uge',   discount: 20 },
    { id: 22, mult: 0.85, label: 'hver-2-uge', discount: 15 },
    { id: 19, mult: 0.95, label: 'hver-4-uge', discount: 5 }
  ];

  // Default: Hver 2 uge
  let selectedFreqIdx = 2;
  let selectedMult = frequencyMap[selectedFreqIdx].mult;
  let selectedFreqId = frequencyMap[selectedFreqIdx].id;
  let selectedLabel = frequencyMap[selectedFreqIdx].label;

  const sqmSliderDesktop = document.getElementById('calc-sqm-desktop');
  const sqmSliderMobile  = document.getElementById('calc-sqm-mobile');
  const sqmValueDesktop  = document.getElementById('calc-sqm-value-desktop');
  const sqmValueMobile   = document.getElementById('calc-sqm-value-mobile');
  const priceSpan        = document.getElementById('calc-price');
  const freqBtns         = document.querySelectorAll('.calc-freq-btn');
  const bookBtn          = document.getElementById('calc-book-btn');

  function getCurrentSlider() {
    return (window.innerWidth <= 768) ? sqmSliderMobile : sqmSliderDesktop;
  }

  function setEditable(span, slider) {
    span.addEventListener('click', function () {
      if (span.querySelector('input')) return;

      const currentVal = span.textContent.trim();
      const input = document.createElement('input');
      input.type = 'number';
      input.value = currentVal;
      input.min = 0;
      input.max = 299;
      input.className = 'calc-sqm-edit-input';
      input.style.width = '80px';

      span.textContent = '';
      span.appendChild(input);
      input.focus();

      function finishEdit() {
        let val = Math.max(0, Math.min(299, parseInt(input.value, 10) || 0));
        slider.value = val;
        span.textContent = val;
        syncAll(val);
      }

      input.addEventListener('blur', finishEdit);
      input.addEventListener('keydown', function(e) {
        if (e.key === 'Enter') input.blur();
        if (e.key === 'Escape') span.textContent = slider.value;
      });
    });
  }

  setEditable(document.getElementById('calc-sqm-value-desktop'), document.getElementById('calc-sqm-desktop'));
  setEditable(document.getElementById('calc-sqm-value-mobile'),  document.getElementById('calc-sqm-mobile'));

  function getTier(sqm) {
    for (let i = 0; i < sqmTiers.length; i++) {
      if (sqm <= sqmTiers[i].max) return sqmTiers[i];
    }
    return sqmTiers[sqmTiers.length - 1];
  }

  // ✅ NEW: base pricing calculator
  function calcBasePrice(sqm) {
    return Math.round(BASE_FEE + (PRICE_PER_M2 * sqm));
  }

  function updatePrice(val) {
    const slider = getCurrentSlider();
    const sqm = val !== undefined ? parseInt(val, 10) : parseInt(slider.value, 10);

    // Set both value fields
    sqmValueDesktop.textContent = sqm;
    sqmValueMobile.textContent = sqm;

    const tier = getTier(sqm);

    // ✅ NEW: price from base model (instead of tier.price)
    const basePrice = calcBasePrice(sqm);

    const totalPrice = Math.round(basePrice * selectedMult);
    priceSpan.textContent = totalPrice;

    // After servicefradrag (26%)
    const afterFradrag = Math.round(totalPrice * 0.74);
    document.getElementById('calc-price-after-tax').textContent =
      afterFradrag.toLocaleString('da-DK') + ' kr.';

    // Store for booking event (unchanged)
    bookBtn.dataset.serviceId = tier.serviceId;
    bookBtn.dataset.sqm = sqm;
  }

  function updateDiscountHighlight(idx) {
    ['discount-once','discount-week','discount-2weeks','discount-4weeks']
      .forEach((id, i) => {
        const el = document.getElementById(id);
        if (el) el.classList.toggle('active', i === idx);
      });
  }

  function syncAll(val) {
    updateDiscountHighlight(selectedFreqIdx);
    updatePrice(val);
  }

  ['input','change'].forEach(evt => {
    sqmSliderDesktop.addEventListener(evt, function() {
      sqmSliderMobile.value = this.value;
      syncAll(this.value);
    });
    sqmSliderMobile.addEventListener(evt, function() {
      sqmSliderDesktop.value = this.value;
      syncAll(this.value);
    });
  });

  freqBtns.forEach((btn, idx) => {
    btn.addEventListener('click', function() {
      freqBtns.forEach(b => b.classList.remove('active'));
      this.classList.add('active');

      selectedMult = parseFloat(this.getAttribute('data-mult'));
      selectedFreqId = frequencyMap[idx].id;
      selectedLabel = this.getAttribute('data-label');
      selectedFreqIdx = idx;

      updateDiscountHighlight(idx);
      updatePrice();
    });
  });

  bookBtn.addEventListener('click', function(e){
  e.preventDefault();

  const sqm = parseInt(getCurrentSlider().value, 10);

  const url =
    'https://renzen.dk/book-rengoering/' +
    '?category_id=3' +
    '&service_id=213' +
    '&pricing_param_quantity=' + sqm +
    '&frequency_id=' + selectedFreqId;

  window.location.href = url;
});


  window.addEventListener('resize', function() {
    const val = getCurrentSlider().value;
    sqmSliderDesktop.value = val;
    sqmSliderMobile.value = val;
    syncAll(val);
  });

  const initialVal = sqmSliderDesktop.value;
  sqmSliderMobile.value = initialVal;
  syncAll(initialVal);
})();
